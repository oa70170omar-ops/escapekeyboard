-- State Control
local autoWinActive = false

-- Targets the absolute reward pad/badge trigger at the very end
local function findTrueRewardPad()
    -- 1. Look specifically for reward/win items inside the workspace
    -- These are the pads that actually give you Trophies/Wins upon contact
    local rewardKeywords = {"Rebirth", "WinPad", "Reward", "TrophyPad", "Collect", "Give", "Badge"}
    
    for i = 1, #rewardKeywords do
        local keyword = rewardKeywords[i]
        for _, obj in ipairs(workspace:GetDescendants()) do
            if obj:IsA("BasePart") and string.find(string.lower(obj.Name), string.lower(keyword)) then
                -- Ignore map boundaries or anti-cheats
                if not string.find(string.lower(obj.Name), "kill") and not string.find(string.lower(obj.Name), "barrier") then
                    return obj
                end
            end
        end
    end

    -- 2. Fallback: Find the part furthest away from spawn that has a TouchTransmitter (a script trigger)
    local bestPart = nil
    local maxDistance = -1
    local originPoint = Vector3.new(0, 0, 0)
    
    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj:IsA("BasePart") and obj:FindFirstChildOfClass("TouchTransmitter") then
            -- Make sure it's not a trap or a player accessory
            if not obj.Parent:FindFirstChild("Humanoid") and not string.find(string.lower(obj.Name), "kill") then
                local dist = (obj.Position - originPoint).Magnitude
                if dist > maxDistance then
                    maxDistance = dist
                    bestPart = obj
                end
            end
        end
    end

    return bestPart
end

local function teleportToFinish()
    local targetPart = findTrueRewardPad()
    if targetPart and player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
        -- Cancel physical momentum so you don't overshoot the pad
        player.Character.HumanoidRootPart.Velocity = Vector3.new(0,0,0)
        -- Snap directly onto the center of the reward platform
        player.Character.HumanoidRootPart.CFrame = targetPart.CFrame + Vector3.new(0, 2, 0)
    end
end

-- Fast Auto-Win loop 
coroutine.wrap(function()
    while true do
        task.wait(0.1) -- Fast snapping loop so it collects rewards instantly
        if autoWinActive then 
            teleportToFinish() 
        end
    end
end)()

player.CharacterAdded:Connect(function(newCharacter) 
    character = newCharacter 
end)
