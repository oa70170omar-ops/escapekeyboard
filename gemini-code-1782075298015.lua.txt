-- MERGED FIXES: Folder Stage Number Check + Absolute Distance Scan
local function findTrueWorld1End()
    local targetPart = nil
    local highestNum = -1
    local maxDistance = -1
    local originPoint = Vector3.new(0, 0, 0)
    
    -- --- FIX 1: FOLDER SCANNING (Choice 2) ---
    local folderNames = {"Stages", "Checkpoints", "Worlds", "Levels"}
    local targetFolder = nil
    
    for i = 1, #folderNames do
        local found = workspace:FindFirstChild(folderNames[i], true)
        if found then
            targetFolder = found
            break
        end
    end
    
    if targetFolder then
        local descendants = targetFolder:GetDescendants()
        for j = 1, #descendants do
            local obj = descendants[j]
            if obj:IsA("BasePart") then
                local num = tonumber(string.match(obj.Name, "%d+"))
                if num and num > highestNum then
                    highestNum = num
                    targetPart = obj
                end
            end
        end
    end
    
    -- --- FIX 2: DISTANCE BACKUP ---
    if not targetPart or highestNum < 5 then 
        local allItems = workspace:GetDescendants()
        for k = 1, #allItems do
            local item = allItems[k]
            if item:IsA("BasePart") and not item.Parent:IsA("Accessory") and not item.Parent:FindFirstChild("Humanoid") then
                local lowerName = string.lower(item.Name)
                if string.find(lowerName, "win") or string.find(lowerName, "finish") or string.find(lowerName, "trophy") or string.match(item.Name, "%d+") then
                    local dist = (item.Position - originPoint).Magnitude
                    if dist > maxDistance then
                        maxDistance = dist
                        targetPart = item
                    end
                end
            end
        end
    end

    return targetPart
end

local function teleportToFinish()
    local targetPart = findTrueWorld1End()
    if targetPart and player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
        -- INSTANT FIX: Pure CFrame assignment with no velocity carry-over
        player.Character.HumanoidRootPart.Velocity = Vector3.new(0,0,0)
        player.Character.HumanoidRootPart.CFrame = targetPart.CFrame + Vector3.new(0, 3, 0)
    end
end

-- Frame-Perfect Auto Win Loop (Instant Response)
coroutine.wrap(function()
    while true do
        task.wait() -- INSTANT FIX: Runs every single game frame instead of waiting 0.3s
        if autoWinActive then 
            teleportToFinish() 
        end
    end
end)()

end)
