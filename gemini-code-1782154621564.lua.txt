local Players = game:GetService("Players")

-- State Control
local autoWinActive = false

local function findAbsoluteFurthestPart()
    local furthestPart = nil
    local maxDistance = 300 -- Ignore anything close to spawn (within 300 studs)
    local centerPoint = Vector3.new(0, 0, 0)
    
    local allItems = workspace:GetDescendants()
    for i = 1, #allItems do
        local item = allItems[i]
        
        -- Make sure it's an actual map part, not a player, accessory, or UI element
        if item:IsA("BasePart") and not item.Parent:IsA("Accessory") and not item.Parent:FindFirstChild("Humanoid") then
            -- Calculate distance ignoring height (Y axis) so skyboxes/falling parts don't break it
            local itemHorizontalPos = Vector3.new(item.Position.X, 0, item.Position.Z)
            local dist = (itemHorizontalPos - centerPoint).Magnitude
            
            if dist > maxDistance then
                -- Skip common UI or lobby parts that might be floating far away
                local lowerName = string.lower(item.Name)
                if not string.find(lowerName, "thumbnail") and not string.find(lowerName, "background") then
                    maxDistance = dist
                    furthestPart = item
                end
            end
        end
    end
    
    return furthestPart
end

local function teleportToFinish()
    local targetPart = findAbsoluteFurthestPart()
    if targetPart and player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
        player.Character.HumanoidRootPart.Velocity = Vector3.new(0, 0, 0)
        -- Instantly snap you to the furthest physical part of the track
        player.Character.HumanoidRootPart.CFrame = targetPart.CFrame + Vector3.new(0, 3, 0)
    end
end

-- Fast Loop Execution
coroutine.wrap(function()
    while true do
        task.wait(0.1)
        if autoWinActive then 
            teleportToFinish() 
        end
    end
end)()

player.CharacterAdded:Connect(function(newCharacter) 
    character = newCharacter 
end)
