local Players = game:GetService("Players")

-- State Control
local autoWinActive = false

-- Hardcoded scan logic prioritizing actual map models over spawn NPCs
local function findActualEndBlock()
    local bestPart = nil
    local highestStageNum = -1
    
    -- 1. Scan typical map hierarchy folders specifically
    local mapContainers = {"Stages", "Checkpoints", "Worlds", "Obby"}
    for i = 1, #mapContainers do
        local container = workspace:FindFirstChild(mapContainers[i], true)
        if container then
            local items = container:GetDescendants()
            for j = 1, #items do
                local obj = items[j]
                if obj:IsA("BasePart") then
                    -- Extract purely stage indicator numbers
                    local num = tonumber(string.match(obj.Name, "%d+"))
                    if num and num > highestStageNum then
                        -- IGNORE SPAWN CHANNELS: Ensure it isn't close to the central hub area coordinates
                        if math.abs(obj.Position.X) > 15 or math.abs(obj.Position.Z) > 15 then
                            highestStageNum = num
                            bestPart = obj
                        end
                    end
                end
            end
        end
    end
    
    -- 2. Direct name match check if the container scan misses
    if not bestPart then
        local winKeywords = {"WinPad", "FinishBlock", "VictoryPad", "EndCheckPoint"}
        for i = 1, #winKeywords do
            local part = workspace:FindFirstChild(winKeywords[i], true)
            if part and part:IsA("BasePart") then
                -- Ignore the donut guy/spawn area platforms entirely
                if part.Position.Y > 5 or math.abs(part.Position.X) > 20 then
                    return part
                end
            end
        end
    end
    
    return bestPart
end

local function teleportToFinish()
    local targetPart = findActualEndBlock()
    if targetPart and player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
        player.Character.HumanoidRootPart.Velocity = Vector3.new(0, 0, 0)
        -- Instantly teleport onto the actual end of the physical track
        player.Character.HumanoidRootPart.CFrame = targetPart.CFrame + Vector3.new(0, 2.5, 0)
    end
end

-- Fast Loop Execution
coroutine.wrap(function()
    while true do
        task.wait(0.1)
        if autoWinActive then 
            teleportToFinish() 
        end
    end
end)()

player.CharacterAdded:Connect(function(newCharacter) 
    character = newCharacter 
end)
