-- Smart Detection Function to locate the Final Stage / Win Area
local function findNewFinalStage()
    local commonNames = {"WinPad", "FinishLine", "Finish", "GiveTrophies", "EndPad", "TrophyPad", "Win"}
    
    -- 1. Scan for exact match names common in this specific game
    for i = 1, #commonNames do
        local name = commonNames[i]
        local found = workspace:FindFirstChild(name, true)
        if found and found:IsA("BasePart") then 
            return found 
        end
    end

    -- 2. Scan folders like "Stages", "Checkpoints", or "Worlds" for the highest level
    local foldersToScan = {"Stages", "Checkpoints", "Worlds"}
    for i = 1, #foldersToScan do
        local folderName = foldersToScan[i]
        local folder = workspace:FindFirstChild(folderName, true)
        if folder then
            local highestPart = nil
            local highestNum = -1
            local descendants = folder:GetDescendants()
            
            for j = 1, #descendants do
                local obj = descendants[j]
                if obj:IsA("BasePart") then
                    local num = tonumber(string.match(obj.Name, "%d+"))
                    if num and num > highestNum then
                        highestNum = num
                        highestPart = obj
                    end
                end
            end
            if highestPart then 
                return highestPart 
            end
        end
    end
    return nil
end

local function teleportToFinish()
    local targetPart = findNewFinalStage()
    if targetPart and player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
        player.Character.HumanoidRootPart.CFrame = targetPart.CFrame + Vector3.new(0, 3, 0)
    end
end

-- Auto Win Loop (Using standard Lua coroutines instead of task.spawn)
coroutine.wrap(function()
    while true do
        wait(0.2) -- Standard Lua wait
        if autoWinActive then 
            teleportToFinish() 
        end
    end
end)()

player.CharacterAdded:Connect(function(newCharacter) 
    character = newCharacter 
end)
